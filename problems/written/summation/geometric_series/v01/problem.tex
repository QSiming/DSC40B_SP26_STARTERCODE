\begin{prob}
    Let $f(n) = \sum_{p=0}^n 3^p$. What is $f$ in $\Theta$ notation?
    
    \begin{soln}
        
        General form of a geometric sum $\displaystyle \sum_{p=0}^n x^p = \frac{1-x^{n+1}}{1-x}$.
        
        Substituting our equation yields $\displaystyle \sum_{p=0}^n 3^p = \frac{1-3^{n+1}}{1-3}$.
        
        Therefore, $\boxed{f(n)=\Theta(3^n)}$ after throwing out the constants.
    
    \end{soln}
    
\end{prob}
