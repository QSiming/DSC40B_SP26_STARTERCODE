    Hint: use the tips for analyzing recursive arguments from lecture.


    \begin{subprobset}
        \begin{subprob}
            In this part, adopt the convention that the sum of the numbers in an empty

            \end{soln}
        \end{subprob}

        \end{subprob}

                \end{soln}
