\begin{prob}

    Suppose a \textbf{directed} graph has 5 nodes and 3 edges. What is the
    greatest possible degree of any node in the graph?

    \inlineresponsebox[.75in]{4}

    \begin{soln}
        Remember that nodes can have self loops, and that the degree of a node
        with only a self-loop is 2. So a node that has a self loop and two
        other edges leaving it has a degree of 4.
    \end{soln}


\end{prob}
