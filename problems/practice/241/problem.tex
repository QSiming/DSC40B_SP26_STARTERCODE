\begin{prob}
    Suppose $G$ is an acyclic directed graph. Let $a$ and $b$ be two nodes of
    $G$ such that $a$ appears before $b$ in \emph{every possible} topological
    sort of $G$.

    True or False: there must be a path from $a$ to $b$.

    \Tf{}

    \begin{soln}
        True
    \end{soln}
\end{prob}
