\begin{prob}
    Consider the function defined below:
    \[
        f(n) =
        \begin{cases}
            3n, & \text{if $n$ is even;}\\
            5n ,& \text{if $n$ is odd.}
        \end{cases}
    \]

    True or False: $f(n) = \Theta(n)$.

    \Tf{}

    \begin{soln}
        True
    \end{soln}

\end{prob}
