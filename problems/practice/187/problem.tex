\begin{prob}
    Let
    \[
        f(n) =
        \frac{n^3 + 10 n}{\sqrt{n} - 100}
        \times
        \frac{ \log(n^2 + 10)} {n(n+1)}
    \]

    Write $f(n)$ in asymptotic notation in the simplest terms possible.

    $f(n) = \Theta($\inlineresponsebox[1in]{$\sqrt n \log n$})

\end{prob}
