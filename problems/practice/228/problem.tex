        def modified_bfs(graph, source):
            """Start a BFS at `source`."""
            status = {node: 'undiscovered' for node in graph.nodes}
            status[source] = 'pending'
            pending = deque([source])

            while pending:
                u = pending.popleft()
                for v in graph.neighbors(u):

                    if status[v] == 'undiscovered':
                        status[v] = 'pending'
                        pending.append(v)

                status[u] = 'visited'
