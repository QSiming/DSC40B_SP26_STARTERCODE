\begin{prob}
    Suppose again that $f_1(n)$ is $O(n^3)$ and $\Omega(n^2)$. Also suppose that $f_2(n)$
    is $O(n^4)$ and $\Omega(n)$.

    Consider the function $g(n) = f_2(n) / f_1(n)$. Give the tightest possible upper bound on $g(n)$:

    $O($\inlineresponsebox[1in]{$n^2$}$)$
\end{prob}
