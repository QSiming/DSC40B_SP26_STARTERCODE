\begin{prob}
    Suppose $f_1(n)$ is $O(n^3)$ and $\Omega(n^2)$. Also suppose that $f_2(n)$
    is $O(n^4)$ and $\Omega(n)$.

    Consider the function $f(n) = f_1(n) + f_2(n)$. True or false: it must be
    the case that $f(n) = \Omega(n^2)$.

    \Tf{}

    \begin{soln}
        True
    \end{soln}
\end{prob}
