        def dfs(graph, u, status=None):
            """Start a DFS at `u`."""
            # initialize status if it was not passed
            if status is None:
                status = {node: 'undiscovered' for node in graph.nodes}

            status[u] = 'pending'
            for v in graph.neighbors(u):
                print("Hello!")
                if status[v] == 'undiscovered':
                    dfs(graph, v, status)
            status[u] = 'visited'
    \end{minted}
