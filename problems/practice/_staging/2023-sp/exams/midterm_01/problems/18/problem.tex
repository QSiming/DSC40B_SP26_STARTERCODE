\begin{prob}
    Consider the function $f(n) = n \times (\sin(n) + 1)$. A plot of this function
    is shown below:

    \begin{center}
        \includegraphics[width=.5\textwidth]{\thisdir/plot.png}
    \end{center}

    True or False: this function is $\Theta(n)$.

    \tF{}

    \begin{soln}
        There is no positive $c$ such that this function is lower-bounded by
        $cn$, and so this function cannot be $\Theta(n)$.
    \end{soln}
\end{prob}
