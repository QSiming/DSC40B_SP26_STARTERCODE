\begin{prob}
    Consider the function $f(n) = \sin(12 n) \cdot \cos(n) + 2$. A plot of this function
    is shown below:

    \begin{center}
        \includegraphics[width=.3\textwidth]{\thisdir/plot.png}
    \end{center}

    True or False: this function is $\Theta(1)$.

    \Tf{}

    \begin{soln}
        This function is upper bounded by 3 and lower bounded by 1 for all $n$, and is therefore $\Theta(1)$.
    \end{soln}
\end{prob}
