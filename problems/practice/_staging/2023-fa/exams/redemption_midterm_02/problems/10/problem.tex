                \node (3) at (1, -1) {$3$};
                \node (4) at (2, -2) {$4$};
                \node (5) at (0, -2) {$5$};
                \node (6) at (-2, -2) {$6$};
                \node (7) at (-3, -3) {$7$};
                \node (8) at (-1, -3) {$8$};
                \node (9) at (1, -3) {$9$};
                \node (10) at (3, -3) {$10$};
                \node (11) at (0, -4) {$11$};
            }

            
            \draw (1) edge[-]  (2);
            \draw (1) edge[-]  (3);
            \draw (2) edge[-]  (6);
            \draw (2) edge[-]  (5);
            \draw (3) edge[-]  (5);
            \draw (3) edge[-]  (4);
