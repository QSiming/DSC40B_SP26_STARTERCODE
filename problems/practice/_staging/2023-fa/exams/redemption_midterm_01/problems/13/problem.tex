\begin{prob}

    Let
    \begin{align*}
        f_1(n) &= \frac{5n^3 + 3n^2 - 10n + 100}{n + 10}\\[2em]
        f_2(n) &= 3n + \sin(\pi n)\\[2em]
        f_3(n) &= \frac{2n^2 + \log n}{(n+5)(n-5)}
    \end{align*}

Define $f(n)  = f_1(n) \times f_2(n) \times f_3(n)$. What is $f(n)$, in asymptotic notation, in
simplest terms possible?

\inlineresponsebox[.75in]{$\Theta(n^3)$}

\end{prob}
