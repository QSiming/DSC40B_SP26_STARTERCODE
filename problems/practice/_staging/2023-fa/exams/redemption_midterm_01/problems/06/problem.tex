\Needspace{2in}
\begin{prob}
    In this problem, assume that $A$ and $B$ are \textbf{sorted} lists,
    each containing $n$ numbers.

    \begin{subprobset}

        \begin{subprob}
            What is the best possible (worst case) time complexity of any
            algorithm for finding the \textbf{smallest} difference between a number in
            $A$ and a number in $B$?

            \inlineresponsebox[.75in]{$\Theta(n)$}
        \end{subprob}

        \begin{subprob}
            What is the best possible (worst case) time complexity of any
            algorithm for finding the \textbf{biggest} difference between a number in
            $A$ and a number in $B$?
        \end{subprob}

        \inlineresponsebox[.75in]{$\Theta(1)$}

    \end{subprobset}

\end{prob}
