\begin{prob}

    Let $G = (V, E)$ be a connected, undirected graph with $|E| = |V| - 1$.
    Suppose $u$ and $v$ are two nodes in $V$.

    True or False: there can be two different simple paths from $u$ to $v$ in $G$.

    \tF{}

    \begin{soln}
        False
    \end{soln}

\end{prob}
